169  template <typename T, typename std::enable_if_t<std::is_integral<T>::value && std::is_unsigned<T>::value, int> = 0>
170  auto DeserializeUnaligned(std::size_t bitSize) -> T
171  {
172  if (mUnalignedBits >= 8)
173  throw SilKitError{"DeserializedUnaligned(): current mUnalignedBits >= 8"};
174  if (bitSize > sizeof(T) * 8)
175  throw SilKitError{"DeserializedUnaligned(): current bitSize > 8 * sizeof(T)"};
176 
177  uint64_t readData = 0;
178  std::size_t readBits = 0;
179 
180  if (mUnalignedBits < bitSize)
181  {
182  auto missingBits = bitSize - mUnalignedBits;
183  auto readBytes = missingBits / 8;
184  auto unalignedBits = missingBits % 8;
185  if (unalignedBits > 0)
186  readBytes += 1;
187  readBits = readBytes * 8;
188 
189  AssertCapacity(readBytes);
190  std::memcpy(&readData, &mBuffer[mReadPos], readBytes);
191  mReadPos += readBytes;
192 
193  mUnalignedData |= (readData << mUnalignedBits);
194 
195  if (mUnalignedBits + readBits > 64u)
196  {
197  readData >>= (64u - mUnalignedBits);
198  readBits = (mUnalignedBits + readBits - 64u);
199  mUnalignedBits = 64u;
200  }
201  else
202  {
203  mUnalignedBits += readBits;
204  readBits = 0;
205  }
206  }
207 
208  T result = static_cast<T>(mUnalignedData);
209  if (bitSize < sizeof(T) * 8)
210  {
211  T mask = (1ull << bitSize) - 1;
212  result &= mask;
213  }
214 
215  mUnalignedData >>= bitSize;
216  mUnalignedBits -= bitSize;
217 
218  if (readBits)
219  {
220  mUnalignedData |= (readData << mUnalignedBits);
221  mUnalignedBits += readBits;
222  }
223  return result;
224  }
225  template <typename T, typename std::enable_if_t<std::is_integral<T>::value && std::is_signed<T>::value, int> = 0>
226  auto DeserializeUnaligned(std::size_t bitSize) -> T
227  {
228  using uint_t = typename std::make_unsigned_t<T>;
229  auto udata = DeserializeUnaligned<uint_t>(bitSize);
230 
231  // Make sure that >> on signed integers is an arithmetic shift. (right shift on signed integers is
232  // implementation defined)
233  static_assert(static_cast<int64_t>((std::numeric_limits<uint64_t>::max)()) >> 63
234  == static_cast<int64_t>((std::numeric_limits<uint64_t>::max)()),
235  "Right shift on signed integers must be an arithmetic shift!");
236 
237  // logic shift everything to MSB
238  udata <<= (sizeof(udata) * 8) - bitSize;
239  // cast to signed integer
240  auto data = static_cast<T>(udata);
241  // arithmetic shift down to LSB for sign extension.
242  data >>= (sizeof(udata) * 8) - bitSize;
243  return data;
244  }

246  template <typename T, typename std::enable_if_t<std::is_integral<T>::value, int> = 0>
247  auto DeserializeAligned(std::size_t numBytes) -> T
248  {
249  Align();
250  AssertCapacity(numBytes);
251 
252  T result;
253  // we copy the "raw" value to the MSB and then shift it down for sign extension
254  std::memcpy(reinterpret_cast<unsigned char*>(&result) + sizeof(T) - numBytes, &mBuffer[mReadPos], numBytes);
255  result >>= (sizeof(T) - numBytes) * 8;
256  mReadPos += numBytes;
257  return result;
258  }
