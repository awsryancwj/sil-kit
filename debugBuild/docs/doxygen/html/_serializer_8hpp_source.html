175  template <typename T,
176  typename std::enable_if<std::is_integral<T>::value && std::is_unsigned<T>::value, int>::type = 0>
177  void SerializeUnaligned(T data, std::size_t bitSize)
178  {
179  if (mUnalignedBits >= 8)
180  throw SilKitError{"SerializedUnaligned(): current mUnalignedBits >= 8"};
181  if (bitSize > sizeof(T) * 8)
182  throw SilKitError{"SerializedUnaligned(): current bitSize > 8 * sizeof(T)"};
183 
184  mUnalignedData |= (static_cast<uint64_t>(data) << mUnalignedBits);
185 
186  auto flushBytes = (mUnalignedBits + bitSize) / 8;
187  auto remainingBits = (mUnalignedBits + bitSize) % 8;
188 
189  if (flushBytes)
190  {
191  auto oldSize = mBuffer.size();
192  mBuffer.resize(oldSize + flushBytes);
193  memcpy(&mBuffer[oldSize], &mUnalignedData, flushBytes);
194 
195  assert(flushBytes <= 8);
196  if (flushBytes == 8)
197  {
198  mUnalignedData = (data >> (64u - mUnalignedBits));
199  }
200  else
201  {
202  mUnalignedData >>= (flushBytes * 8);
203  }
204  }
205  assert(remainingBits < 8);
206  mUnalignedBits = static_cast<uint8_t>(remainingBits);
207  }
