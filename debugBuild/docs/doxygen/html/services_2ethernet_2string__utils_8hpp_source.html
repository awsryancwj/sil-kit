118 std::ostream& operator<<(std::ostream& out, const EthernetFrame& msg)
119 {
120  if (msg.raw.size() == 0)
121  {
122  return out
123  << "EthernetFrame{size=0}";
124  }
125  else
126  {
127  using EthernetMac = std::array<uint8_t, 6>;
128 
129  out << "EthernetFrame{size=" << msg.raw.size();
130  if (msg.raw.size() >= 2 * sizeof(EthernetMac))
131  {
132  EthernetMac destinationMac;
133  EthernetMac sourceMac;
134  std::copy(
135  msg.raw.begin(),
136  msg.raw.begin() + sizeof(EthernetMac), destinationMac.begin());
137  std::copy(
138  msg.raw.begin() + sizeof(EthernetMac),
139  msg.raw.begin() + 2 * sizeof(EthernetMac), sourceMac.begin());
140 
141  out << ", src = " << Util::AsHexString(sourceMac).WithSeparator(":")
142  << ", dst=" << Util::AsHexString(destinationMac).WithSeparator(":");
143  }
144  return out
145  << ", data=[" << Util::AsHexString(msg.raw).WithSeparator(" ").WithMaxLength(8)
146  << "]}";
147  }
148 }
