164 std::ostream& operator<<(std::ostream& out, const CanFrame& msg)
165 {
166  out << "Can::CanFrame{"
167  << ", canId=" << msg.canId
168  << ", flags=";
169 
170  auto printFlag = [firstFlag = true, &out, &msg](const CanFrameFlag flag) mutable {
171  if (!(msg.flags & static_cast<CanFrameFlagMask>(flag)))
172  {
173  return;
174  }
175 
176  if (firstFlag)
177  {
178  firstFlag = false;
179  }
180  else
181  {
182  out << ",";
183  }
184 
185  out << flag;
186  };
187 
188  // print all flags (including potentially unknown ones)
189  out << '[';
190  for (uint32_t bit = 0; bit != 32u; ++bit)
191  {
192  printFlag(static_cast<CanFrameFlag>(BIT(bit)));
193  }
194  out << ']';
195 
196  return out
197  << ", dlc=" << static_cast<uint32_t>(msg.dlc)
198  << ", sdt=" << static_cast<uint32_t>(msg.sdt)
199  << ", vcid=" << static_cast<uint32_t>(msg.vcid)
200  << ", af=" << static_cast<uint32_t>(msg.af)
201  << ", data=[" << Util::AsHexString(msg.dataField).WithSeparator(" ").WithMaxLength(8)
202  << "], data.size=" << msg.dataField.size() << "}";
203 }
