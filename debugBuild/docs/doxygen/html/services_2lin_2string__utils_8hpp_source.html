246 std::ostream& operator<<(std::ostream& out, const LinFrame& frame)
247 {
248  //instead of ios::copyfmt (which set badbit) we use a temporary stream
249  std::stringstream buf;
250  buf
251  << "Lin::LinFrame{id=" << static_cast<uint16_t>(frame.id)
252  << ", cs=" << to_string(frame.checksumModel)
253  << ", dl=" << static_cast<uint16_t>(frame.dataLength)
254  << ", d={" << Util::AsHexString(frame.data).WithSeparator(" ")
255  << "}}";
256  return out << buf.str();
257 }

259 std::ostream& operator<<(std::ostream& out, const LinControllerConfig& controllerConfig)
260 {
261  out << "Lin::LinControllerConfig{br=" << controllerConfig.baudRate

264  if (controllerConfig.frameResponses.size() > 0)
265  {
266  out << static_cast<uint16_t>(controllerConfig.frameResponses[0].frame.id);

268  {
269  out << "," << static_cast<uint16_t>(controllerConfig.frameResponses[i].frame.id);
270  }
