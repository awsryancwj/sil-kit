285 std::ostream& operator<<(std::ostream& out, const FlexrayHeader& header)
286 {
287  using FlagMask = FlexrayHeader::FlagMask;
288  using Flag = FlexrayHeader::Flag;
289  return out
290  << "Flexray::FlexrayHeader{f=["
291  << ((header.flags & static_cast<FlagMask>(Flag::SuFIndicator)) ? "U" : "-")
292  << ((header.flags & static_cast<FlagMask>(Flag::SyFIndicator)) ? "Y" : "-")
293  << ((header.flags & static_cast<FlagMask>(Flag::NFIndicator)) ? "-" : "N")
294  << ((header.flags & static_cast<FlagMask>(Flag::PPIndicator)) ? "P" : "-")
295  << "],s=" << header.frameId
296  << ",l=" << (uint32_t)header.payloadLength
297  << ",crc=" << std::hex << header.headerCrc << std::dec
298  << ",c=" << (uint32_t)header.cycleCount
299  << "}";
300 }

328 std::ostream& operator<<(std::ostream& out, const FlexrayFrameEvent& msg)
329 {
330  auto timestamp = std::chrono::duration_cast<std::chrono::duration<double, std::milli>>(msg.timestamp);
331  out << "Flexray::FlexrayFrameEvent{"
332  << "ch=" << msg.channel
333  << ", " << msg.frame.header
334  << " @" << timestamp.count() << "ms";
335  if (msg.frame.header.flags & static_cast<FlexrayHeader::FlagMask>(FlexrayHeader::Flag::NFIndicator))
336  {
337  // if payload is valid, provide it as hex dump
338  out << ", payload="
339  << Util::AsHexString(msg.frame.payload).WithSeparator(" ").WithMaxLength(msg.frame.header.payloadLength);
340  }
341  return out << "}";
342 }
