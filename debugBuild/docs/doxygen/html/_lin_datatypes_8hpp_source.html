318 inline auto GoToSleepFrame() -> LinFrame
319 {
320  LinFrame frame;
321  frame.id = 0x3c;
322  frame.checksumModel = LinChecksumModel::Classic;
323  frame.dataLength = 8;
324  frame.data = {0x0, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff};
