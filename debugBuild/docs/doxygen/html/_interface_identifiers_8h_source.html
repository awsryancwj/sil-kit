55 #define SK_ID_GET_SERVICE(ID) \
56  ((ID >> 40) & 0xFF)
57 #define SK_ID_GET_DATATYPE(ID)\
59  ((ID >> 32) & 0xFF)
60 #define SK_ID_GET_VERSION(ID)\
62  ((ID >> 24) & 0xFF)
63 
64 #define SK_ID_MAKE(SERVICE_NAME, DATATYPE_NAME) \
65  (uint64_t)(\
66  ((uint64_t)83 /*S*/ << 56)\
67  | ((uint64_t)75 /*K*/ << 48)\
68  | ((uint64_t)(SK_ID_SERVICE_ ## SERVICE_NAME & 0xFF) << 40)\
69  | ((uint64_t)(DATATYPE_NAME ## _DATATYPE_ID & 0xff) << 32)\
70  | ((uint64_t)(DATATYPE_NAME ## _VERSION & 0xff) << 24)\
71  | 0x0)
72 
73 #define SK_ID_IS_VALID(ID)\
74  (\
75  (SK_ID_GET_SERVICE(ID) > SK_ID_SERVICE_START)\
76  && (SK_ID_GET_SERVICE(ID) < SK_ID_SERVICE_END)\
77  && (SK_ID_GET_DATATYPE(ID) != SK_INVALID_DATATYPE_ID)\
78  && (SK_ID_GET_VERSION(ID) != SK_INVALID_VERSION)\
79  && ((ID >> 56 & 0xff) == /*S*/ 83)\
80  && ((ID >> 48 & 0xff) == /*K*/ 75)\
81  )
