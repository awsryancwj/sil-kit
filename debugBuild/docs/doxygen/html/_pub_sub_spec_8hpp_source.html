67 void PubSubSpec::AddLabel(const SilKit::Services::MatchingLabel& label)
68 {
69  if (label.kind != MatchingLabel::Kind::Mandatory && label.kind != MatchingLabel::Kind::Optional)
70  {
71  throw ConfigurationError(
72  "SilKit::Services::MatchingLabel must specify a SilKit::Services::MatchingLabel::Kind.");
73  }
74 
75  for (auto& it : _labels)
76  {
77  if (it.key == label.key)

79  it = label;

81  }
