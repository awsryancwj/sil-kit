57  inline void to_ostream(std::ostream& out) const
58  {
59  if (_maxLength <= 0)
60  return;
61 
62  const auto begin = _iterable.begin();
63  const auto end = begin + _maxLength;
64 
65  //NB, we use a temporary stream because we don't want to modify the internal format state of out
66  std::ostringstream oss;
67 
68  oss << std::hex << std::setfill('0')
69  << std::setw(2) << static_cast<uint16_t>(*begin);
70 
71  if (_maxLength > 1)
72  {
73  std::for_each(begin + 1,
74  end,
75  [&oss, this](auto chr)
76  {
77  oss << _separator << std::setw(2) << static_cast<uint16_t>(chr);
78  }
79  );
80  }
81  if (_maxLength < _iterable.end() - _iterable.begin())
82  {
83  oss << _separator << "...";
84  }
85  out << oss.str();
86  }
87 
88 private:
89  using difference_type = decltype(std::declval<IterableT>().end()
90  - std::declval<IterableT>().begin());
91 
92  const IterableT& _iterable;
93  difference_type _maxLength = (std::numeric_limits<difference_type>::max)();
94  std::string _separator = "";
