152 std::ostream& operator<<(std::ostream& out, const ParticipantStatus& status)
153 {
154  std::time_t enterTime = std::chrono::system_clock::to_time_t(status.enterTime);
155  std::tm tmBuffer;
156 #if defined(_WIN32)
157  localtime_s(&tmBuffer, &enterTime);
158 #else
159  localtime_r(&enterTime, &tmBuffer);
160 #endif
161 
162  out << "Orchestration::ParticipantStatus{" << status.participantName
163  << ", State=" << status.state
164  << ", Reason=" << status.enterReason
165  << ", Time=" << std::put_time(&tmBuffer, "%FT%T")
166  << "}";
